Mission 1 Solution: A* Search

import heapq

def a_star_search(graph, heuristic, start, goal):
    # Priority queue stores tuples of (f_cost, node, path, g_cost).
    # f_cost = g_cost (actual cost) + h_cost (estimated cost).
    # We use a priority queue to always explore the node with the lowest f_cost first.
    open_list = [(heuristic[start], start, [start], 0)]
    
    # A set to keep track of nodes we've already visited to prevent cycles.
    visited = set()

    # Loop as long as there are nodes to explore.
    while open_list:
        # Get the node with the lowest f_cost from the priority queue.
        f_cost, current_node, path, g_cost = heapq.heappop(open_list)

        # If we've already processed this node, skip it.
        if current_node in visited:
            continue
        
        # Mark the current node as visited.
        visited.add(current_node)

        # If we've reached the goal, we're done! Return the path and total cost.
        if current_node == goal:
            return path, g_cost
        
        # Explore the neighbors of the current node.
        for neighbor, cost in graph.get(current_node, []):
            if neighbor not in visited:
                # Calculate the new costs for the path through the current node to the neighbor.
                new_g_cost = g_cost + cost
                new_f_cost = new_g_cost + heuristic[neighbor]
                
                # Add the neighbor to the priority queue to be explored later.
                heapq.heappush(open_list, (new_f_cost, neighbor, path + [neighbor], new_g_cost))

    # If the loop finishes and we haven't found the goal, no path exists.
    return None, None

# --- Mission 1 Data ---
graph = {'S':[('A',3),('B',2)],'A':[('C',4),('D',1)],'B':[('E',3),('F',1)],'C':[],'D':[('G',5)],'E':[('H',5)],'F':[('I',2),('G',3)],'G':[],'H':[],'I':[]}
heuristic = {'S':5,'A':7,'B':4,'C':99,'D':4,'E':6,'F':2,'G':0,'H':99,'I':99}
start_node = 'S'
goal_node = 'G'

# --- Execute and Print Solution ---
path, cost = a_star_search(graph, heuristic, start_node, goal_node)
print("--- A* Search Solution ---")
if path:
    print(f"Path found: {path}")
    print(f"Total cost: {cost}")
else:
    print("No path found.")

Mission 2 Solution: Hill Climbing

def hill_climbing(landscape, start_index):
    current_index = start_index
    max_index = len(landscape) - 1

    # Loop until we reach a peak (a point where no neighbor is higher).
    while True:
        current_value = landscape[current_index]
        best_neighbor_index = current_index
        best_neighbor_value = current_value

        # Check the neighbor to the right.
        right_index = current_index + 1
        if right_index <= max_index and landscape[right_index] > best_neighbor_value:
            best_neighbor_value = landscape[right_index]
            best_neighbor_index = right_index

        # Check the neighbor to the left.
        left_index = current_index - 1
        if left_index >= 0 and landscape[left_index] > best_neighbor_value:
            best_neighbor_value = landscape[left_index]
            best_neighbor_index = left_index
        
        # If the best neighbor is no better than the current position, we've found a peak.
        if best_neighbor_index == current_index:
            return current_index, current_value
        
        # Otherwise, move to the better neighbor and continue climbing.
        current_index = best_neighbor_index

# --- Mission 2 Data ---
rooftops = [2, 5, 8, 12, 11, 9, 15, 14, 10]
start_pos_1 = 0
start_pos_2 = 8

# --- Execute and Print Solutions ---
print("\n--- Hill Climbing Solutions ---")
index1, value1 = hill_climbing(rooftops, start_pos_1)
print(f"From start {start_pos_1}: Vantage point found at index {index1} with height {value1}")

index2, value2 = hill_climbing(rooftops, start_pos_2)
print(f"From start {start_pos_2}: Vantage point found at index {index2} with height {value2}")

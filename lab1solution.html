Q1.5 Leap Year Check

import calendar

year = int(input("Enter year to check: "))

# Method 1: Using Calendar Module
if calendar.isleap(year):
    print(f"{year} is a Leap Year (using module)")
else:
    print(f"{year} is NOT a Leap Year (using module)")

# Method 2: Logic
if (year % 4 == 0 and year % 100 != 0) or (year % 400 == 0):
    print(f"{year} is a Leap Year (using logic)")
else:
    print(f"{year} is NOT a Leap Year (using logic)")

my_str = input("Enter a string: ")

my_str = my_str.casefold()

Q3.2 Fibonacci Series

n_terms = int(input("How many terms? "))

n1, n2 = 0, 1
count = 0

if n_terms <= 0:
    print("Please enter a positive integer")
elif n_terms == 1:
    print("Fibonacci sequence upto", n_terms, ":")
    print(n1)
else:
    print("Fibonacci sequence:")
    while count < n_terms:
        print(n1)
        nth = n1 + n2
        # update values
        n1 = n2
        n2 = nth
        count += 1

Part 4: Functions & Lists

Q4.1 Factorial Function

def factorial(n):
    if n < 0:
        return "Undefined for negative numbers"
    elif n == 0 or n == 1:
        return 1
    else:
        return n * factorial(n-1)

num = int(input("Enter a number: "))
print(f"Factorial of {num} is {factorial(num)}")

Q4.2 Compare Lists

def compare_lists(l1, l2):
    # Python lists compare element-by-element by default
    if l1 == l2:
        return True
    else:
        return False

list1 = [1, 2, 3]
list2 = [1, 2, 3]
list3 = [3, 2, 1]

print(f"List 1 vs List 2: {compare_lists(list1, list2)}") # True
print(f"List 1 vs List 3: {compare_lists(list1, list3)}") # False

Q4.3 List Statistics (No Built-ins)

# Input list
user_input = input("Enter numbers separated by space: ")
numbers = [int(x) for x in user_input.split()]

sum_val = 0
count = 0
max_val = numbers[0]

for num in numbers:
    sum_val += num
    count += 1
    if num > max_val:
        max_val = num

average = sum_val / count

print(f"Sum: {sum_val}")
print(f"Average: {average}")
print(f"Maximum: {max_val}")

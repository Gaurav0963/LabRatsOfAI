Q1. Array Initialization

Concepts: arange, ones, dtype

import numpy as np

# 1D array of integers from 0 to 9
arr_1d = np.arange(10)
print("1D Array:", arr_1d)

# 3x3 matrix of True values
# dtype=bool forces the 1s to be interpreted as True
bool_matrix = np.ones((3, 3), dtype=bool)
print("Boolean Matrix:\n", bool_matrix)

Q2. Sequence Generation

Concepts: linspace vs arange

# 10 points between 0 and 1 inclusive
# linspace is better when you need a specific number of points
lin_arr = np.linspace(0, 1, 10)
print("Linspace:", lin_arr)

# Comparison: arange takes a 'step' size, not 'count'
# arange(0, 1.1, 0.1) might result in floating point errors
print("Arange:", np.arange(0, 1.1, 0.1))

Q3. Array Attributes

Concepts: shape, ndim, size

# Random 3x4x2 array
arr = np.random.random((3, 4, 2))

print("Dimensions (ndim):", arr.ndim) # Output: 3
print("Shape:", arr.shape)            # Output: (3, 4, 2)
print("Size:", arr.size)              # Output: 24 (3*4*2)
print("Data Type:", arr.dtype)        # Output: float64

Q4. Reshaping

Concepts: reshape, flatten

arr = np.arange(20)

# Reshape to 4x5
reshaped = arr.reshape((4, 5))
print("Reshaped (4x5):\n", reshaped)

# Flatten back to 1D
flattened = reshaped.flatten()
print("Flattened:", flattened)

Q5. Slicing Matrix

Concepts: slicing [start:stop]

# 5x5 Identity Matrix
identity = np.eye(5)

# Extract central 3x3 (rows 1-3, cols 1-3)
# Remember: index 4 is exclusive
sub_matrix = identity[1:4, 1:4]

print("Original:\n", identity)
print("Central 3x3:\n", sub_matrix)

Q6. Boolean Indexing

Concepts: masking, replacement

arr = np.random.randint(1, 51, 20)
print("Original:", arr)

# Replace odd numbers with -1
# arr % 2 != 0 creates a boolean mask
arr[arr % 2 != 0] = -1

print("Modified:", arr)

Q7. Where Condition

Concepts: np.where

a = np.array([1, 2, 3, 4, 5])
b = np.array([5, 4, 3, 2, 1])

# Where a > b, take from a, else take from b
result = np.where(a > b, a, b)

print("A:", a)
print("B:", b)
print("Result (Max of element-wise):", result)

Q8. Axis Aggregations

Concepts: sum(axis), max(axis)

arr = np.random.randint(0, 10, (3, 4))
print("Array:\n", arr)

# Sum of columns (axis 0 - collapsing rows)
col_sum = arr.sum(axis=0)
print("Sum of columns:", col_sum)

# Max of rows (axis 1 - collapsing columns)
row_max = arr.max(axis=1)
print("Max of rows:", row_max)

Q9. Broadcasting

Concepts: keepdims, element-wise math

arr = np.random.random((3, 4))

# Calculate mean of each row
# keepdims=True ensures result is (3,1) instead of (3,)
# This allows it to broadcast against (3,4)
means = arr.mean(axis=1, keepdims=True)

normalized = arr - means
print("Original Mean (approx):", arr.mean())
print("New Row Means (should be 0):", normalized.mean(axis=1))

Q10. Standard Deviation

Concepts: np.random.randn, mean, std

# randn generates from "Standard Normal" (mean=0, std=1)
data = np.random.randn(1000)

print(f"Mean: {data.mean():.4f}") # Should be close to 0
print(f"Std Dev: {data.std():.4f}") # Should be close to 1

Q11. Matrix Multiplication

Concepts: dot, matmul (@)

mat1 = np.random.random((5, 3))
mat2 = np.random.random((3, 2))

result = np.dot(mat1, mat2)

Q12. Stacking Arrays

Concepts: vstack, hstack

a = np.array([1, 2, 3])
b = np.array([4, 5, 6])

# Vertical Stack (one on top of other)
v_stack = np.vstack((a, b))
print("Vertical Stack:\n", v_stack)
# Shape: (2, 3)

# Horizontal Stack (side by side)
h_stack = np.hstack((a, b))
print("Horizontal Stack:\n", h_stack)
# Shape: (6,)

Q13. Unique Items

Concepts: np.unique

arr = np.random.randint(0, 11, 50)

# return_counts=True gives frequency
values, counts = np.unique(arr, return_counts=True)

print("Unique Values:", values)
print("Counts:       ", counts)

Q14. Linear Algebra (Inverse)

Concepts: linalg.det, linalg.inv

mat = np.random.randint(1, 10, (3, 3))
print("Matrix:\n", mat)

det = np.linalg.det(mat)
print(f"Determinant: {det:.2f}")

if det != 0:
    inv = np.linalg.inv(mat)
    print("Inverse:\n", inv)
else:
    print("Matrix is singular, no inverse.")

Q15. Set Operations

a = np.array([1,2,3,2,3,4,3,4,5,6])

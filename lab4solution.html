Q1. Inspection & Structure

import pandas as pd
import seaborn as sns

# Load dataset
df = sns.load_dataset('iris')

# Inspect
print(df.head())
print("\nInfo:")
print(df.info()) # Shows types and memory usage

Q2. Statistical Summary

# Describe and transpose for readability
stats = df.describe().T
print(stats)

Q3. Class Balance

# Count species
print(df['species'].value_counts())

# Check nulls
print("\nMissing Values:", df.isnull().sum().sum())

Q4. Aggregations

# Groupby and multiple aggregations
grouped = df.groupby('species')['sepal_length'].agg(['mean', 'median'])
print(grouped)

Q5. Array Conversion (NumPy)

import numpy as np

sepal_length_arr = df['sepal_length'].values # or .to_numpy()

print("Variance:", np.var(sepal_length_arr))
print("Std Dev:", np.std(sepal_length_arr))

Q6. Standardization

sw = df['sepal_width'].values
mean_sw = np.mean(sw)
std_sw = np.std(sw)

# Vectorized operation
df['sepal_width_z'] = (sw - mean_sw) / std_sw

print(df[['sepal_width', 'sepal_width_z']].head())

Q7. Filtering with NumPy

# np.where(condition, value_if_true, value_if_false)
df['petal_size_category'] = np.where(df['petal_length'] > 4.5, 'Large', 'Small')

print(df[['petal_length', 'petal_size_category']].sample(5))

Q8. Matrix Operations

# Drop non-numeric columns
numeric_data = df.select_dtypes(include=[np.number])

# Pandas correlation
print("Pandas Corr:\n", numeric_data.corr())

# NumPy correlation (requires transposing so rows are variables)
print("\nNumPy Corr:\n", np.corrcoef(numeric_data.T))

Q9. Scatter Plot (Matplotlib)

import matplotlib.pyplot as plt

plt.figure(figsize=(8, 6))
plt.scatter(df['sepal_length'], df['sepal_width'], c='blue', alpha=0.5)
plt.title("Sepal Dimensions")
plt.xlabel("Sepal Length")
plt.ylabel("Sepal Width")
plt.show()

Q10. Histograms (Matplotlib)

plt.figure(figsize=(8, 6))
plt.hist(df['petal_length'], bins=15, color='green', alpha=0.7)
plt.title("Distribution of Petal Length")
plt.xlabel("Length")
plt.ylabel("Frequency")
plt.show()

Q11. Boxplot (Seaborn)

plt.figure(figsize=(8, 6))
sns.boxplot(x='species', y='petal_width', data=df)
plt.title("Petal Width by Species")
plt.show()

Q12. Violin Plot

plt.figure(figsize=(8, 6))
sns.violinplot(x='species', y='sepal_length', data=df)
plt.title("Sepal Length Distribution (Violin)")
plt.show()

Q13. Pairplot

sns.pairplot(df, hue='species')
plt.show()

Q14. Heatmap

# Drop non-numeric for correlation
corr = df.drop(columns=['species', 'petal_size_category', 'sepal_width_z'], errors='ignore').corr()

plt.figure(figsize=(8, 6))
sns.heatmap(corr, annot=True, cmap='coolwarm')
plt.title("Feature Correlation")
plt.show()
# Observation: Petal Length and Petal Width are strongly correlated.

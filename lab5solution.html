Q1

Rotting Oranges (BFS)

Approach: We use a Queue to perform Breadth-First Search. We first enqueue all initially rotten oranges. Then, we process them level by level (minute by minute), infecting adjacent fresh oranges. We keep a count of fresh oranges to easily check if any are left at the end.

from collections import deque

def orangesRotting(grid):
    rows, cols = len(grid), len(grid[0])
    queue = deque()
    fresh_count = 0
    
    # Step 1: Initialize the queue with all rotten oranges
    # and count the fresh ones.
    for r in range(rows):
        for c in range(cols):
            if grid[r][c] == 2:
                queue.append((r, c, 0)) # (row, col, minutes_passed)
            elif grid[r][c] == 1:
                fresh_count += 1
                
    # If there are no fresh oranges, it takes 0 minutes.
    if fresh_count == 0:
        return 0
        
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    minutes_passed = 0
    
    # Step 2: Perform BFS
    while queue:
        r, c, mins = queue.popleft()
        minutes_passed = mins
        
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            # If adjacent cell is within bounds and has a fresh orange
            if 0 <= nr < rows and 0 <= nc < cols and grid[nr][nc] == 1:
                # Rot the orange
                grid[nr][nc] = 2
                fresh_count -= 1
                queue.append((nr, nc, mins + 1))
                
    # Step 3: Check if any fresh oranges are left
    if fresh_count == 0:
        return minutes_passed
    else:
        return -1

# Test the function
grid = [[2,1,1],[1,1,0],[0,1,1]]
print("Minimum minutes:", orangesRotting(grid)) # Output: 4
Q2

Word Search (DFS)

Approach: We iterate through each cell. If the letter matches the start of the word, we trigger a Depth-First Search with backtracking. We maintain a set of visited coordinates to avoid reusing cells in the current path. If a path fails, we backtrack by removing the cell from visited.

def exist(board, word):
    rows, cols = len(board), len(board[0])
    path = set() # To keep track of visited cells in current DFS path
    
    def dfs(r, c, i):
        # Base Case 1: We found all characters of the word
        if i == len(word):
            return True
            
        # Base Case 2: Out of bounds, wrong letter, or cell already visited
        if (r < 0 or c < 0 or 
            r >= rows or c >= cols or 
            word[i] != board[r][c] or 
            (r, c) in path):
            return False
            
        # Add current cell to path before exploring neighbors
        path.add((r, c))
        
        # Explore all 4 adjacent directions
        res = (dfs(r + 1, c, i + 1) or 
               dfs(r - 1, c, i + 1) or 
               dfs(r, c + 1, i + 1) or 
               dfs(r, c - 1, i + 1))
               
        # Backtrack: Remove cell from path so it can be used in other DFS branches
        path.remove((r, c))
        return res

    # Start DFS from every cell
    for r in range(rows):
        for c in range(cols):
            if dfs(r, c, 0): 
                return True
                
    return False

# Test the function
board = [
    ["A","B","C","E"],
    ["S","F","C","S"],
    ["A","D","E","E"]
]
print("Exists 'ABCCED':", exist(board, "ABCCED")) # Output: True
print("Exists 'ABCB':", exist(board, "ABCB"))     # Output: False
